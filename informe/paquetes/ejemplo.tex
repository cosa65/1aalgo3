
\algoritmo{encontrarPos}{in exploradoras : Ronda,in exp : char}{res : int}{
  \Var{i : int}
  \complejidad{O(1)}
  \State $i \larr 0$
  \complejidad{O(1)}
  \While{i $<$ exploradoras.size()}
  \complejidad{O($e$)}
  \If{exp == exploradoras[i]}
  \complejidad{O(1)}
  \State $res \larr i$
  \complejidad{O(1)}
  \EndIf
  \State $i \larr i+1$
  \complejidad{O(1)}
  \EndWhile
} {O($e$)}
